n=int(input())
a=list(map(int,input().split()))
if n<3:
    print("false")
else:
    i=1
    while i<n and a[i]>a[i-1]:
        i=i+1
    if i==1 or i==n:
        print("true")
    else:
        while i<n and a[i]<a[i-1]:
            i=i+1
        if i==n:
            print("true")
        else:
            print("false")
